Collection Types

Tuple

Stored contiguously in memory!

{1, "aloha", :hey}

Use case: pattern matching

{:ok, file} = File.open("file.txt")

List

Lists are linked lists!

[1, "aloha", :hey]
[head | tail] #=> [1 | ["aloha", :hey]]
[1 | ["aloha" | [:hey | []]]]

Keyword list

Key must be an atom!

[a: 1, b: "aloha", c: :hey]
[{:a, 1}, {:b, "aloha"}, {:c, :hey}]

Use case: options to functions

Supervisor.start_link(children, strategy: :one_for_one,
  name: Discordia.Supervisor)

Map

Order is not guaranteed!

%{:a => 1, 2 => "aloha", "cool" => :hey}
%{name: "Ricardo", username: "rkma"}

Match operator

left = right

status = {:started, :normal}

case status do
  {:plus_hold, ^value, _} ->
    {:ok, status}
  {:plus_hold, status_value, _} ->
    {:error, "Player must play a #{status_value} card."}
  {:ended, _} ->
    {:error, "Game is over."}
  {_, _} ->
    {:ok, status}
end
